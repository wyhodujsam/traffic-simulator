```java
// The ONLY path that preserves raw ReaderWay tags + exposes node tags.
import com.graphhopper.reader.osm.WaySegmentParser;
import com.graphhopper.storage.BaseGraph;
import com.graphhopper.storage.Directory;
import com.graphhopper.storage.RAMDirectory;
import com.graphhopper.util.PointList;
import com.graphhopper.reader.ReaderWay;

Directory dir = new RAMDirectory();
BaseGraph graph = new BaseGraph.Builder(0).setDir(dir).create();

WaySegmentParser parser = new WaySegmentParser.Builder(graph.getNodeAccess(), dir)
    .setWayFilter(way -> way.hasTag("highway"))
    .setEdgeHandler((from, to, pointList, way, nodeTags) -> {
        // from, to         : int tower-node IDs (graph-local, NOT OSM ids)
        // pointList        : PointList with lat/lon for all (tower + pillar) nodes
        // way              : ReaderWay — full OSM tags via way.getTag("key")
        // nodeTags         : List<Map<String, Object>> — tags of each node along segment
        //                    Index 0 == `from` tower's tags
        //                    Index last == `to` tower's tags
    })
    .setWorkerThreads(1)
    .build();

parser.readOSM(osmFile);   // java.io.File
```
