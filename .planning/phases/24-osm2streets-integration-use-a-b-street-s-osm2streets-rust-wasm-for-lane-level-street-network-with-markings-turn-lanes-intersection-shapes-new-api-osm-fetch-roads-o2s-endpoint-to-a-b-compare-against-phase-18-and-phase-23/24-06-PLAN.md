      ```tsx
      const handleFetchRoadsO2s = useCallback(async () => {
        if (!bbox) return;
        setState('loading');
        setLoadingMessage('Fetching roads via osm2streets...');
        setError(null);
        try {
          const response = await fetch('/api/osm/fetch-roads-o2s', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ south: bbox.south, west: bbox.west, north: bbox.north, east: bbox.east }),
          });
          if (!response.ok) {
            const errJson = await response.json().catch(() => ({}));
            throw new Error(errJson.error || `HTTP ${response.status}`);
          }
          const data: MapConfigData = await response.json();
          const roadCount = data.roads?.length ?? 0;
          const intersectionCount = data.intersections?.length ?? 0;
          setResult({ roadCount, intersectionCount });
          setResultOrigin('osm2streets');
          setMapConfig(data);
          setState('result');
        } catch (e) {
          setError((e as Error).message);
          setState('error');
        }
      }, [bbox]);
      ```
      Mirror the EXACT error-handling structure of `handleFetchRoadsGh` — do not invent new patterns. If Phase 23 clone uses a different helper (e.g. a `postBbox` util), use the same helper.
