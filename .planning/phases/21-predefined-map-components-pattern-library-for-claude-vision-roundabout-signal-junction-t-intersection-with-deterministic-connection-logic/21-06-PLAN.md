```java
@SpringBootTest
@EnabledIfSystemProperty(named = "vision.harness", matches = "true")
class VisionComparisonHarness {
    @Autowired ClaudeVisionService phase20;
    @Autowired ComponentVisionService phase21;
    @Autowired MapValidator validator;

    @Test
    void compareOnRoundaboutFixture() throws Exception {
        Path fixture = Path.of("/tmp/roundabout-test.png");
        Assumptions.assumeTrue(Files.exists(fixture),
            "Fixture /tmp/roundabout-test.png not present; harness skipped.");

        byte[] png = Files.readAllBytes(fixture);

        MapConfig a = safeCall(() -> phase20.analyzeImageBytes(png));
        MapConfig b = safeCall(() -> phase21.analyzeImageBytes(png));

        Path out = Path.of("target/vision-comparison");
        Files.createDirectories(out);

        ObjectMapper mapper = new ObjectMapper().enable(SerializationFeature.INDENT_OUTPUT);
        if (a != null) mapper.writeValue(out.resolve("free-form.json").toFile(), a);
        if (b != null) mapper.writeValue(out.resolve("components.json").toFile(), b);

        String report = buildDiffReport(a, b);
        Files.writeString(out.resolve("diff.md"), report);

        // Do NOT fail the test on divergence — the harness is diagnostic, not regressive.
        // Log summary to test output instead.
        System.out.println(report);
    }
}
```
