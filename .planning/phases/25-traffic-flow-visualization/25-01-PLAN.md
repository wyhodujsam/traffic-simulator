```java
package com.trafficsimulator.engine;

import static org.assertj.core.api.Assertions.assertThat;

import java.lang.reflect.Field;
import java.util.ArrayDeque;
import java.util.Deque;

import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

class VehicleSpawnerTickWindowTest {

    private VehicleSpawner spawner;

    @BeforeEach
    void setUp() {
        spawner = new VehicleSpawner();
    }

    @Test
    void tickWindow_evictsEntriesOlderThanSixtySecondsOfTicks() throws Exception {
        seedDeque(0L, 100L, 500L, 1200L, 1300L);
        // currentTick=1300, cutoff = 1300 - 60*20 = 100. Entries < 100 are evicted (only tick 0).
        // Remaining: 100, 500, 1200, 1300 = 4 entries.
        assertThat(spawner.getThroughput(1300)).isEqualTo(4);
    }

    @Test
    void tickWindow_emptyAfterReset() throws Exception {
        seedDeque(50L, 100L);
        spawner.reset();
        assertThat(spawner.getThroughput(1000)).isZero();
    }

    @Test
    void tickWindow_neverNegativeOnFreshSpawner() {
        assertThat(spawner.getThroughput(0)).isZero();
    }

    @Test
    void tickWindow_sameTickWindow_inclusive() throws Exception {
        seedDeque(100L, 200L);
        // cutoff = 1300 - 1200 = 100. Entry at 100 is NOT < 100, so it stays.
        assertThat(spawner.getThroughput(1300)).isEqualTo(2);
        // cutoff = 1301 - 1200 = 101. Entry at 100 IS < 101, so evicted.
        assertThat(spawner.getThroughput(1301)).isEqualTo(1);
    }

    @SuppressWarnings("unchecked")
    private void seedDeque(Long... ticks) throws Exception {
        Field f = VehicleSpawner.class.getDeclaredField("despawnTicks");
        f.setAccessible(true);
        Deque<Long> deque = (Deque<Long>) f.get(spawner);
        deque.clear();
        for (Long t : ticks) deque.addLast(t);
    }
}
```

```java
import com.trafficsimulator.model.Intersection;
import com.trafficsimulator.model.IntersectionType;
import com.trafficsimulator.model.Lane;
import com.trafficsimulator.model.Road;
import com.trafficsimulator.model.RoadNetwork;
import com.trafficsimulator.model.Vehicle;

/**
 * WAVE-0 SPIKE — answers RESEARCH.md Open Question Q1 / Pitfall #2.
 *
 * <p>Builds an 8-segment ring (8 nodes, 8 roads, each segment 250m, 2 lanes), assigns each
 * intersection IntersectionType.PRIORITY, primes 8 vehicles uniformly spaced, runs 100 ticks of
 * physics + intersection transfers, and asserts all 8 vehicles are still alive AND have advanced.
 *
 * <p>If this test FAILS with PRIORITY, repeat with IntersectionType.NONE — that is the recommended
 * fallback per Pitfall #2. Document the result in 25-WAVE0-SPIKE-RESULT.md so Plan 03 picks the
 * right intersection type for ring-road.json.
 */
class RingRoadPriorityYieldSpikeTest {

    private static final double SEGMENT_LENGTH = 250.0;
    private static final double SPEED_LIMIT = 22.2;
    private static final int LANE_COUNT = 2;
    private static final int VEHICLE_COUNT = 8;
    private static final int TICK_COUNT = 100;

    @Test
    void ringWithPriorityDoesNotStall() {
        RoadNetwork network = buildRingNetwork(IntersectionType.PRIORITY);
        primeUniformVehicles(network, VEHICLE_COUNT);

        // Run 100 ticks. We do not invoke the full Spring pipeline; we only need to know if the
        // intersection layer would block motion. The simplest proxy: count vehicles before vs after.
        int before = countVehicles(network);
        runTicks(network, TICK_COUNT);
        int after = countVehicles(network);

        assertThat(after).as("vehicle count after ring run with PRIORITY").isEqualTo(before);
        assertThat(totalAdvancedDistance(network)).as("total advance distance").isGreaterThan(0.0);
    }

    @Test
    void ringWithNoneDoesNotStall_fallbackOption() {
        RoadNetwork network = buildRingNetwork(IntersectionType.NONE);
        primeUniformVehicles(network, VEHICLE_COUNT);
        runTicks(network, TICK_COUNT);
        assertThat(countVehicles(network)).isEqualTo(VEHICLE_COUNT);
        assertThat(totalAdvancedDistance(network)).isGreaterThan(0.0);
    }

    // --- Helpers ----------------------------------------------------------

    private RoadNetwork buildRingNetwork(IntersectionType type) {
        // Build 8 nodes around a circle of radius r = 2000 / (2*PI)
        double radius = 2000.0 / (2.0 * Math.PI);
        double cx = 500.0;
        double cy = 500.0;

        Map<String, Road> roads = new LinkedHashMap<>();
        Map<String, Intersection> intersections = new LinkedHashMap<>();

        // Build roads r0..r7 connecting node n_i -> n_(i+1 mod 8)
        for (int i = 0; i < 8; i++) {
            String roadId = "r" + i;
            Road road = new Road(
                    roadId, "Seg " + i,
                    cx + radius * Math.cos(2 * Math.PI * i / 8),
                    cy + radius * Math.sin(2 * Math.PI * i / 8),
                    cx + radius * Math.cos(2 * Math.PI * (i + 1) / 8),
                    cy + radius * Math.sin(2 * Math.PI * (i + 1) / 8),
                    SEGMENT_LENGTH,
                    SPEED_LIMIT,
                    LANE_COUNT);
            // Lanes are typically initialised by the Road constructor — check signature.
            roads.put(roadId, road);
        }
        // Build 8 intersections wiring inbound r_(i-1) -> outbound r_i
        for (int i = 0; i < 8; i++) {
            String nodeId = "n" + i;
            Intersection ixtn = new Intersection(nodeId, type, 0.0);
            ixtn.getInboundRoadIds().add("r" + ((i + 7) % 8));
            ixtn.getOutboundRoadIds().add("r" + i);
            intersections.put(nodeId, ixtn);
        }
        return new RoadNetwork("ring-spike", roads, intersections, List.of(), List.of());
    }

    private void primeUniformVehicles(RoadNetwork network, int count) {
        List<Road> roads = new ArrayList<>(network.getRoads().values());
        for (int i = 0; i < count; i++) {
            Road r = roads.get(i % roads.size());
            Lane lane = r.getLanes().get(0);
            Vehicle v = Vehicle.builder()
                    .id(UUID.randomUUID().toString())
                    .position(0.0)
                    .speed(SPEED_LIMIT * 0.8)
                    .acceleration(0.0)
                    .lane(lane)
                    .length(4.5)
                    .v0(33.3)
                    .aMax(1.4)
                    .b(2.0)
                    .s0(2.0)
                    .timeHeadway(1.5)
                    .spawnedAt(0L)
                    .build();
            lane.addVehicle(v);
        }
    }

    private void runTicks(RoadNetwork network, int ticks) {
        // Minimal tick: advance position by speed * dt (0.05s) and let vehicles wrap to the next
        // road via a simple intersection-transfer proxy (advance to next road when position >= length).
        double dt = 0.05;
        for (int t = 0; t < ticks; t++) {
            for (Road road : network.getRoads().values()) {
                for (Lane lane : road.getLanes()) {
                    for (Vehicle v : new ArrayList<>(lane.getVehiclesView())) {
                        double newPos = v.getPosition() + v.getSpeed() * dt;
                        if (newPos >= road.getLength()) {
                            // Transfer to next road in ring
                            int idx = Integer.parseInt(road.getId().substring(1));
                            Road nextRoad = network.getRoads().get("r" + ((idx + 1) % 8));
                            Lane nextLane = nextRoad.getLanes().get(0);
                            v.updatePhysics(newPos - road.getLength(), v.getSpeed(), 0.0);
                            v.setLane(nextLane);
                            lane.removeVehicle(v);
                            nextLane.addVehicle(v);
                        } else {
                            v.updatePhysics(newPos, v.getSpeed(), 0.0);
                        }
                    }
                }
            }
        }
    }

    private int countVehicles(RoadNetwork network) {
        int total = 0;
        for (Road r : network.getRoads().values())
            for (Lane l : r.getLanes()) total += l.getVehiclesView().size();
        return total;
    }

    private double totalAdvancedDistance(RoadNetwork network) {
        double sum = 0.0;
        for (Road r : network.getRoads().values())
            for (Lane l : r.getLanes())
                for (Vehicle v : l.getVehiclesView()) sum += v.getPosition();
        return sum;
    }
}
```
