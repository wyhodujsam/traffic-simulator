package com.trafficsimulator.integration;

import static org.assertj.core.api.Assertions.assertThat;

import java.io.IOException;
import java.nio.file.Files;
import java.nio.file.Path;
import java.util.UUID;

import org.junit.jupiter.api.AfterEach;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.test.context.SpringBootTest;
import org.springframework.test.context.DynamicPropertyRegistry;
import org.springframework.test.context.DynamicPropertySource;

import com.trafficsimulator.engine.CommandDispatcher;
import com.trafficsimulator.engine.SimulationEngine;
import com.trafficsimulator.engine.command.SimulationCommand;
import com.trafficsimulator.replay.ReplayLogger;
import com.trafficsimulator.config.MapLoader;

/**
 * DET-01 (HEADLINE) + DET-02. Same seed → byte-identical NDJSON.
 *
 * <p>Scope: byte-identity is asserted ONLY against the NDJSON replay log content (per
 * RESEARCH.md Open Questions Q4/Q6 RESOLVED — STOMP frames contain wall-clock-driven
 * timestamps and KPI sub-samples and are explicitly out of scope).
 *
 * <p>File naming: this class uses `*Test.java` (not `*IT.java`) because backend/pom.xml
 * has no Failsafe plugin; Surefire's default patterns would silently exclude `*IT.java`.
 */
@SpringBootTest
class DeterminismTest {

    static Path tmpDir;

    @DynamicPropertySource
    static void configureReplay(DynamicPropertyRegistry r) throws IOException {
        tmpDir = Files.createTempDirectory("phase25-determinism-");
        r.add("simulator.replay.enabled", () -> "true");
        r.add("simulator.replay.directory", () -> tmpDir.toString());
    }

    @Autowired SimulationEngine engine;
    @Autowired CommandDispatcher dispatcher;
    @Autowired ReplayLogger replayLogger;
    @Autowired MapLoader mapLoader;

    @BeforeEach
    void setup() throws Exception {
        // Load ring-road
        var loaded = mapLoader.loadFromClasspath("maps/ring-road.json");
        engine.setRoadNetwork(loaded.network());
    }

    @AfterEach
    void teardown() throws IOException { replayLogger.close(); }

    @Test
    void sameSeedSameLog() throws Exception {
        Path p1 = runOnce(42L, 200L);
        Path p2 = runOnce(42L, 200L);
        long mismatch = Files.mismatch(p1, p2);
        assertThat(mismatch).as("DET-01 HEADLINE: same seed must yield byte-identical NDJSON").isEqualTo(-1L);
    }

    @Test
    void differentSeedDifferentLog() throws Exception {
        Path p1 = runOnce(42L, 200L);
        Path p2 = runOnce(43L, 200L);
        long mismatch = Files.mismatch(p1, p2);
        assertThat(mismatch).as("DET-02: different seeds must yield different NDJSON").isNotEqualTo(-1L);
    }

    /** Helper: re-load scenario, start with a specific seed, run N ticks, capture the file. */
    private Path runOnce(long seed, long ticks) throws Exception {
        // Reload to reset state
        var loaded = mapLoader.loadFromClasspath("maps/ring-road.json");
        engine.setRoadNetwork(loaded.network());
        engine.getTickCounter().set(0L);

        // Start with explicit seed
        dispatcher.dispatch(new SimulationCommand.Start(seed));
        // Run fast for N ticks (auto-stops, blocking on completion via wait loop)
        dispatcher.dispatch(new SimulationCommand.RunForTicksFast(ticks));

        // Wait for fast run to complete (max 30s)
        long deadline = System.currentTimeMillis() + 30_000L;
        while (engine.isFastMode() && System.currentTimeMillis() < deadline) {
            Thread.sleep(50);
        }
        assertThat(engine.isFastMode()).as("fast run should complete within 30s").isFalse();

        Path p = replayLogger.getPath();
        assertThat(p).isNotNull();
        replayLogger.close();
        // Move to a known unique name so subsequent runs don't overwrite
        Path renamed = p.resolveSibling(seed + "-" + UUID.randomUUID() + ".ndjson");
        Files.move(p, renamed);
        return renamed;
    }
}

/** DET-06: RUN_FOR_TICKS=N auto-stops after N ticks. */
@SpringBootTest
class RunForTicksTest {

    @Autowired SimulationEngine engine;
    @Autowired CommandDispatcher dispatcher;
    @Autowired MapLoader mapLoader;

    @Test
    void autoStopAtTargetTick() throws Exception {
        var loaded = mapLoader.loadFromClasspath("maps/ring-road.json");
        engine.setRoadNetwork(loaded.network());
        engine.getTickCounter().set(0L);

        dispatcher.dispatch(new SimulationCommand.Start(42L));
        long startTick = engine.getTickCounter().get();
        dispatcher.dispatch(new SimulationCommand.RunForTicksFast(50L));

        long deadline = System.currentTimeMillis() + 10_000L;
        while (engine.isFastMode() && System.currentTimeMillis() < deadline) {
            Thread.sleep(20);
        }

        assertThat(engine.isFastMode()).isFalse();
        assertThat(engine.getStatus()).isEqualTo(SimulationStatus.STOPPED);
        assertThat(engine.getTickCounter().get())
            .as("DET-06: tick counter at or just past startTick + 50")
            .isBetween(startTick + 50L, startTick + 60L);  // small tolerance for last-tick race
    }
}

@SpringBootTest
class FastModeParityTest {

    static Path tmpDir;

    @DynamicPropertySource
    static void configureReplay(DynamicPropertyRegistry r) throws IOException {
        tmpDir = Files.createTempDirectory("phase25-fastparity-");
        r.add("simulator.replay.enabled", () -> "true");
        r.add("simulator.replay.directory", () -> tmpDir.toString());
    }

    @Autowired SimulationEngine engine;
    @Autowired CommandDispatcher dispatcher;
    @Autowired ReplayLogger replayLogger;
    @Autowired MapLoader mapLoader;

    @Test
    void sameSeedFastVsSlowMatch() throws Exception {
        Path fast = runWith(42L, 50L, true);
        Path slow = runWith(42L, 50L, false);
        long mismatch = Files.mismatch(fast, slow);
        assertThat(mismatch)
            .as("DET-07: FAST and wall-clock modes must produce identical NDJSON for same seed")
            .isEqualTo(-1L);
    }

    private Path runWith(long seed, long ticks, boolean fast) throws Exception {
        var loaded = mapLoader.loadFromClasspath("maps/ring-road.json");
        engine.setRoadNetwork(loaded.network());
        engine.getTickCounter().set(0L);

        dispatcher.dispatch(new SimulationCommand.Start(seed));
        if (fast) {
            dispatcher.dispatch(new SimulationCommand.RunForTicksFast(ticks));
        } else {
            dispatcher.dispatch(new SimulationCommand.RunForTicks(ticks));
        }
        long deadline = System.currentTimeMillis() + 30_000L;
        while ((engine.isFastMode() || engine.getAutoStopTick() >= 0L) && System.currentTimeMillis() < deadline) {
            Thread.sleep(20);
        }
        Path p = replayLogger.getPath();
        replayLogger.close();
        Path renamed = p.resolveSibling((fast ? "fast-" : "slow-") + UUID.randomUUID() + ".ndjson");
        Files.move(p, renamed);
        return renamed;
    }
}

import static org.assertj.core.api.Assertions.assertThat;

import java.util.List;

import org.junit.jupiter.api.Test;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.test.context.SpringBootTest;

import com.trafficsimulator.config.MapLoader;
import com.trafficsimulator.dto.SegmentKpiDto;
import com.trafficsimulator.dto.SimulationStateDto;
import com.trafficsimulator.engine.CommandDispatcher;
import com.trafficsimulator.engine.SimulationEngine;
import com.trafficsimulator.engine.command.SimulationCommand;
import com.trafficsimulator.engine.IVehicleSpawner;
import com.trafficsimulator.scheduler.SnapshotBuilder;

@SpringBootTest
class RingRoadTest {

    @Autowired SimulationEngine engine;
    @Autowired CommandDispatcher dispatcher;
    @Autowired MapLoader mapLoader;
    @Autowired SnapshotBuilder snapshotBuilder;
    @Autowired IVehicleSpawner spawner;

    @Test
    void ringDoesNotStall_RING02() throws Exception {
        loadAndStart(42L);
        dispatcher.dispatch(new SimulationCommand.RunForTicksFast(100L));
        waitFastDone();

        SimulationStateDto state = currentState();
        assertThat(state.getVehicles().size())
            .as("RING-02: 80 vehicles must still be present after 100 ticks (no stall)")
            .isEqualTo(80);
        assertThat(state.getVehicles().stream().anyMatch(v -> v.getPosition() > 0))
            .as("at least one vehicle has moved").isTrue();
    }

    @Test
    void steadyStateLosCorD_RING03() throws Exception {
        loadAndStart(43L);
        dispatcher.dispatch(new SimulationCommand.RunForTicksFast(150L));    // pre-perturbation
        waitFastDone();

        SimulationStateDto state = currentState();
        List<SegmentKpiDto> segs = state.getStats().getSegmentKpis();
        assertThat(segs).as("RING-03: segmentKpis must be populated (sub-sample tick % 5 == 0)").isNotEmpty();
        long countCorD = segs.stream().filter(s -> "C".equals(s.getLos()) || "D".equals(s.getLos())).count();
        assertThat(countCorD)
            .as("RING-03: at least 6 of 8 segments at LOS C or D in steady state")
            .isGreaterThanOrEqualTo(6);
    }

    @Test
    void perturbationProducesLosF_RING04() throws Exception {
        loadAndStart(44L);
        dispatcher.dispatch(new SimulationCommand.RunForTicksFast(500L));    // past perturbation @ tick=200, +60 duration
        waitFastDone();

        SimulationStateDto state = currentState();
        List<SegmentKpiDto> segs = state.getStats().getSegmentKpis();
        boolean anyF = segs.stream().anyMatch(s -> "F".equals(s.getLos()));
        assertThat(anyF)
            .as("RING-04: after perturbation, at least one segment must hit LOS F by tick 500")
            .isTrue();
    }

    private void loadAndStart(long seed) throws Exception {
        var loaded = mapLoader.loadFromClasspath("maps/ring-road.json");
        engine.setRoadNetwork(loaded.network());
        engine.getTickCounter().set(0L);
        dispatcher.dispatch(new SimulationCommand.Start(seed));
    }

    private void waitFastDone() throws InterruptedException {
        long deadline = System.currentTimeMillis() + 60_000L;
        while (engine.isFastMode() && System.currentTimeMillis() < deadline) {
            Thread.sleep(50);
        }
    }

    private SimulationStateDto currentState() {
        return snapshotBuilder.buildSnapshot(new SnapshotBuilder.SnapshotConfig(
            engine.getRoadNetwork(),
            engine.getTickCounter().get(),
            engine.getStatus().name(),
            spawner,
            engine.getRoadNetwork().getId(),
            null));
    }
}

import { test, expect } from '@playwright/test';

/**
 * UI-03: Phantom-jam visual proof. Loads ring-road, runs 500 ticks fast, expands diagnostics
 * panel, samples space-time canvas, asserts a continuous low-speed (red-tinted) vertical band
 * of length corresponding to ≥100 m of road appears.
 *
 * Per CONTEXT.md "Specific Ideas": "user can see the wave" — formalised here as
 * "at least one column of the space-time canvas contains a contiguous vertical run of
 * red-dominant pixels of length ≥80 px" (canvas height = 200 px, y axis = 250 m → 80px ≈ 100m).
 *
 * Trigger mechanism: relies on `window.__SIM_SEND_COMMAND__` exposed in dev mode by
 * useWebSocket.ts (gated by import.meta.env.DEV). NO `test.skip()` branch — the dev shim
 * is part of the contract.
 */
test.describe('Phase 25 — Phantom jam visual proof (UI-03)', () => {

  test('ring-road perturbation produces visible jam wave on space-time diagram', async ({ page }) => {
    await page.goto('/');
    await page.waitForLoadState('networkidle');

    // 1. Wait for the dev shim to be installed by useWebSocket.ts
    await page.waitForFunction(() => typeof (window as any).__SIM_SEND_COMMAND__ === 'function', { timeout: 10_000 });

    // 2. Load ring-road, START with seed=42, RUN_FOR_TICKS_FAST=500
    await page.evaluate(async () => {
      const send = (window as any).__SIM_SEND_COMMAND__;
      // LOAD_CONFIG with the ring-road map (the backend serves it from classpath)
      send({ type: 'LOAD_MAP', mapId: 'ring-road' });
    });
    // small wait for map load
    await page.waitForTimeout(500);

    await page.evaluate(async () => {
      const send = (window as any).__SIM_SEND_COMMAND__;
      send({ type: 'START', seed: 42 });
      send({ type: 'RUN_FOR_TICKS_FAST', ticks: 500 });
    });

    // 3. Expand diagnostics panel
    const toggleButton = page.getByRole('button', { name: /show diagnostics/i });
    await toggleButton.click();
    await expect(page.getByTestId('space-time-canvas')).toBeVisible();

    // 4. Wait for the fast run to complete + STOMP frames to populate canvas
    await page.waitForTimeout(8000);

    // 5. Sample canvas pixel data
    const canvasHandle = await page.getByTestId('space-time-canvas').elementHandle();
    if (!canvasHandle) throw new Error('canvas not found');

    const { hasJam, tallestRun } = await canvasHandle.evaluate((canvas: HTMLCanvasElement) => {
      const ctx = canvas.getContext('2d');
      if (!ctx) return { hasJam: false, tallestRun: 0 };
      const w = canvas.width;
      const h = canvas.height;
      const data = ctx.getImageData(0, 0, w, h).data;
      let tallest = 0;
      // For each column, find longest vertical run of red-dominant pixels (r > 200, g < 100)
      for (let x = 0; x < w; x++) {
        let run = 0;
        for (let y = 0; y < h; y++) {
          const idx = (y * w + x) * 4;
          const r = data[idx], g = data[idx + 1];
          if (r > 200 && g < 100) {
            run++;
            if (run > tallest) tallest = run;
          } else {
            run = 0;
          }
        }
      }
      // 80 px ≈ 100 m on a 200px / 250m y-axis (scale ~0.8 px/m)
      return { hasJam: tallest >= 80, tallestRun: tallest };
    });

    console.log('UI-03: tallest red run = ' + tallestRun + ' px');
    expect(hasJam, `UI-03: expected at least 80px (≈100m) tall jam band, got ${tallestRun}px`).toBe(true);
  });
});
