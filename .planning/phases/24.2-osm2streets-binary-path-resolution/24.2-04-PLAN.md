    if (!response.ok) {

        errorMessage = (parsed && typeof parsed.error === 'string')
          ? parsed.error
          : (text || `HTTP ${response.status}`);
      } catch {
        errorMessage = text || `HTTP ${response.status}`;
      }
      throw new Error(errorMessage);
    }
